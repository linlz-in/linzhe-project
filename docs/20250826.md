* 3.作业：求每一天登录的新老用户数
  SELECT
  login_date,
  -- 统计新用户数：登录日期 = 首次登录日期
  SUM(CASE WHEN login_date = first_login_date THEN 1 ELSE 0 END) AS new_user_count,
  -- 统计老用户数：登录日期 > 首次登录日期
  SUM(CASE WHEN login_date > first_login_date THEN 1 ELSE 0 END) AS old_user_count
  FROM (
  -- 获取每日去重的登录用户，并关联其首次登录日期
  SELECT DISTINCT
  l.login_date,
  l.user_id,
  -- 子查询2：计算每个用户的首次登录日期
  (SELECT MIN(login_date)
  FROM fact_log
  WHERE user_id = l.user_id
  ) AS first_login_date
  FROM fact_log l
  ) AS daily_login_info
  -- 按日期分组统计
  GROUP BY login_date
  -- 按日期升序排列
  ORDER BY login_date;
